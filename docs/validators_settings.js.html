/**
 * @fileoverview Configuration validation and default settings management.
 * @module validators/settings
 */

import { ConfigError } from "../utils/error.js";

/**
 * Default user settings that can be overridden by configuration.
 * @constant {Object}
 * @property {string} docSelector - CSS selector for document elements.
 * @property {Array<string>} searchableAttributes - Attributes to index.
 * @property {Array<string>} stopWords - Words excluded from indexing.
 * @property {Array<string>} attributesToRetrieve - Attributes to return in results.
 * @property {number} minCharsFor1Typo - Minimum characters to allow 1 typo.
 * @property {number} minCharsFor2Typos - Minimum characters to allow 2 typos.
 * @property {Array<Object>} customRanking - Custom ranking rules.
 * @property {string} searchBarSelector - CSS selector for search input.
 */
const userSettings = {
  docSelector: ".card",
  searchableAttributes: ["title", "description"],
  stopWords: ["a", "and", "the", "f", "for"],
  attributesToRetrieve: ["*"],
  minCharsFor1Typo: 4,
  minCharsFor2Typos: 6,
  customRanking: [
    { attribute: "popularity", direction: 1 },
    { attribute: "price", direction: -1 },
  ],
  searchBarSelector: "#searchBar",
};

/**
 * Internal engine defaults that cannot be overridden.
 * @constant {Object}
 * @property {number} disableTypoToleranceBeforeQueryLength - Minimum query length for fuzzy matching.
 */
const engineDefaults = {
  disableTypoToleranceBeforeQueryLength: 3,
};

/**
 * Validates user configuration and merges with default settings.
 * @param {Object} config - User-provided configuration object.
 * @param {string} [config.docSelector] - CSS selector for document elements.
 * @param {Array<string>} [config.searchableAttributes] - Attributes to index.
 * @param {Array<string>} [config.stopWords] - Words to exclude from indexing.
 * @param {Array<string>} [config.attributesToRetrieve] - Attributes to return.
 * @param {number} [config.minCharsFor1Typo] - Min chars for 1 typo tolerance.
 * @param {number} [config.minCharsFor2Typos] - Min chars for 2 typos tolerance.
 * @param {Array<Object>} [config.customRanking] - Custom ranking rules.
 * @param {string} [config.searchBarSelector] - CSS selector for search input.
 * @returns {Object} Object containing merged userSettings and engineDefaults.
 * @throws {ConfigError} If config is not provided or contains invalid keys.
 * @throws {Error} If typo tolerance settings are invalid.
 * @example
 * const { userSettings, engineDefaults } = validateAndExportSettings({
 *   docSelector: ".product",
 *   minCharsFor1Typo: 4,
 *   minCharsFor2Typos: 7
 * });
 */
export function validateAndExportSettings(config) {
  if (!config) {
    throw new ConfigError("Please provide a config");
  }
  if (config.minCharsFor1Typo < 3) {
    throw new ConfigError("minCharsFor1Typo must be 3 or greater");
  }
  if (config.minCharsFor1Typo == config.minCharsFor2Typos) {
    throw new Error("minCharsFor1Typo and minCharsFor2Typos can't be equal");
  }
  if (config.minCharsFor1Typo > config.minCharsFor2Typos) {
    throw new Error("minCharsFor1Typo must be less than minCharsFor2Typos");
  }

  const validConfigKeys = Object.keys(userSettings);
  const userConfigKeys = Object.keys(config);

  for (const userConfigKey of userConfigKeys) {
    if (!validConfigKeys.includes(userConfigKey)) {
      throw new ConfigError(
        `"${userConfigKey}" is not a valid config attribute`,
      );
    }
    userSettings[userConfigKey] = config[userConfigKey];
  }

  return { userSettings, engineDefaults };
}
